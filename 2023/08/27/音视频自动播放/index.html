音视频自动播放

发表于2023-08-27|更新于2023-08-27
|阅读量:

html

<div class="container">
  <video controls width="500" autoplay class="player">
    <source src="./demo.mp4" />
  </video>
  <button class="btn">开始播放</button>
</div>

css

.container {
  display: flex;
  justify-content: center;
  align-items: center;
  position: relative;
  
  .btn {
    position: absolute;
    background-color: khaki;
    border: unset;
    border-radius: 15px;
    width: 100px;
    height: 30px;
    cursor: pointer;
  }
}

js

1.互动后出声

const btn = document.querySelector('.btn');
const vdo = document.querySelector('.player');
// 1.互动后出声
function play() {
  vdo.muted = true; //静音
  vdo.play();
  const ctx = new AudioContext();
  console.log('====================================');
  console.log('ctx', ctx);
  console.log('====================================');
  const canAutoPLay = ctx?.state === 'running';
  ctx.close();
  if (canAutoPLay) {
    vdo.muted = false;
    btn.style.display = 'none';
    btn.removeEventListener('click', play);
  } else {
    btn.style.display = 'block';
    btn.addEventListener('click', play);
  }
}

play()

   async function play() {
    try {
      const res = await vdo.play();
      console.log('====================================');

      btn.addEventListener('click', play);
    } catch (error) {

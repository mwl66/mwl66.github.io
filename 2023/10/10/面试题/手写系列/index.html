版本号比较

  // 1.0.0.0
  // 1.0
  // 1.0.1
  // 2.0
  // // 1.0.0.0和1.0相等，1.0小于1.0.1，1.0.1小于2.0
  // // v1大于v2返回1，v1小于v2返回-1，相等返回0
  const versionCompare = (version1, version2) => {
  const v1Parts = version1.split('.').map(Number);
  const v2Parts = version2.split('.').map(Number);
  const maxLength = Math.max(v1Parts.length, v2Parts.length);

  for (let i = 0; i < maxLength; i++) {
    const v1Part = v1Parts[i] || 0;
    const v2Part = v2Parts[i] || 0;

    if (v1Part < v2Part) {
      return -1;
    } else if (v1Part > v2Part) {
      return 1;
    }
  }

  return 0; // 两个版本号相等
};
console.log(versionCompare("1.0.0.0", "1.0"));
console.log(versionCompare("1.0", "1.0.1"));
console.log(versionCompare("1.0.1", "2.0"));

函数柯里化，实现如下效果

// console.log(add(1,2,3))
// //  打印6
// console.log(add(1)(2,3))
// //  打印6
// console.log(add(1)(3)(2))
// //  打印6
// console.log(add(1,2)(3))

function add(...args) {
  if (args.length >= 3) {
    return args.reduce((total, num) => total + num, 0);
  } else {
    return function (...nextArgs) {
      return add(...args.concat(nextArgs));
    };
  }
}
